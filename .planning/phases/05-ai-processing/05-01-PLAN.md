    ```typescript
    import type { RewrittenContent } from '../types/post';

    const GEMINI_ENDPOINT = 'https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent';

    interface GeminiRequest {
      contents: Array<{
        parts: Array<{
          text: string;
        }>;
      }>;
    }

    /**
     * Rewrite news content to engaging Vietnamese social post using Gemini 2.0 Flash
     */
    export async function rewriteWithGemini(
      title: string,
      summary: string,
      source: string,
      apiKey: string
    ): Promise<RewrittenContent> {
      const prompt = buildPrompt(title, summary, source);
      
      const requestBody: GeminiRequest = {
        contents: [{
          parts: [{ text: prompt }]
        }]
      };

      const response = await fetch(`${GEMINI_ENDPOINT}?key=${apiKey}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(requestBody)
      });

      if (!response.ok) {
        const error = await response.text();
        throw new Error(`Gemini API error: ${response.status} - ${error}`);
      }

      const result = await response.json();
      const rewrittenText = extractTextFromResponse(result);
      
      // Extract hashtags and hook from the generated text
      const { content, hashtags, hook } = parseGeneratedContent(rewrittenText);

      return {
        originalTitle: title,
        originalSummary: summary,
        rewrittenText: content,
        hashtags,
        hook,
        wordCount: countWords(content)
      };
    }

    function buildPrompt(title: string, summary: string, source: string): string {
      return `Bạn là một biên tập viên tin tức chuyên nghiệp. Hãy viết lại tin tức sau thành bài đăng Facebook tiếng Việt hấp dẫn:

Tin gốc:
- Tiêu đề: ${title}
- Nội dung: ${summary}
- Nguồn: ${source}

Yêu cầu:
1. Viết hook hấp dẫn 1-2 câu đầu
2. Nội dung chính 150-200 chữ, giữ nguyên thông tin chính xác
3. Thêm 2-3 hashtags phù hợp
4. Format phù hợp đọc trên mobile

Viết ngay bài đăng, không giải thích thêm.`;
    }

    function extractTextFromResponse(response: any): string {
      try {
        return response.candidates?.[0]?.content?.parts?.[0]?.text || '';
      } catch {
        throw new Error('Failed to parse Gemini response');
      }
    }

    function parseGeneratedContent(text: string): { 
      content: string; 
      hashtags: string[]; 
      hook: string 
    } {
      // Extract hashtags (lines starting with #)
      const hashtagRegex = /#[\w\u00C0-\u024F]+/g;
      const hashtags = (text.match(hashtagRegex) || []).slice(0, 3);
      
      // Remove hashtags from content for clean output
      const contentWithoutHashtags = text.replace(hashtagRegex, '').trim();
      
      // First 1-2 sentences as hook
      const sentences = contentWithoutHashtags.split(/[.!?]+/);
      const hook = sentences.slice(0, 2).join('. ').trim() + '.';
      
      // Content without hashtags
      const content = contentWithoutHashtags;
      
      return { content, hashtags, hook };
    }

    function countWords(text: string): number {
      return text.trim().split(/\s+/).filter(w => w.length > 0).length;
    }
    ```

    **Important**: 
    - Use Gemini 2.0 Flash (free tier, 500 requests/day)
    - Parse Vietnamese hashtags properly with unicode support
    - Ensure word count is 150-200 words
