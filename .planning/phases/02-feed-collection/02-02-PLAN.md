    const HOT_KEYWORDS = [
      'AI', 'GPT', 'OpenAI', 'Google', 'Apple', 'Meta', 'Microsoft',
      'iPhone', 'Samsung', 'Tesla', 'Trump', 'Putin', 'Biden',
      'Nvidia', 'chip', 'Semicon', 'Trump', ' tariffs', 'trade war',
      'Bitcoin', 'Crypto', 'USD', 'Fed', 'interest rate'
    ];

    const TOPIC_KEYWORDS = {
      tech: ['AI', 'GPT', 'tech', 'software', 'app', 'phone', 'computer'],
      world: ['trump', 'putin', 'biden', 'war', 'military', 'politics'],
      business: ['stock', 'market', 'economy', 'fed', 'rate', 'crypto'],
      science: ['space', 'NASA', 'research', 'climate', 'energy']
    };

    function scoreItem(item) {
      let score = 0;

      // 1. Hot keywords (40 điểm)
      const titleLower = item.title.toLowerCase();
      const summaryLower = item.summary?.toLowerCase() || '';
      for (const kw of HOT_KEYWORDS) {
        if (titleLower.includes(kw.toLowerCase())) {
          score += 10;
        }
        if (summaryLower.includes(kw.toLowerCase())) {
          score += 5;
        }
      }

      // 2. Số lượng cùng topic trong ngày (30 điểm)
      let topicCount = 0;
      for (const [topic, keywords] of Object.entries(TOPIC_KEYWORDS)) {
        for (const kw of keywords) {
          if (titleLower.includes(kw.toLowerCase())) {
            topicCount++;
          }
        }
      }
      score += Math.min(topicCount * 5, 30);

      // 3. Engagement giả lập (30 điểm)
      // RSS thường không có engagement, tính theo:
      // - Có media/image: +10
      // - Title có số: +5 (thường là tin thời sự)
      // - Độ dài summary > 100: +5
      if (item.summary?.length > 100) score += 5;
      if (/\d+/.test(titleLower)) score += 5;
      if (item.source === 'BBC' || item.source === 'Reuters') score += 10;

      return score;
    }

    function scoreAll(items) {
      return items.map(item => ({
        ...item,
        hotScore: scoreItem(item)
      }));
    }

    module.exports = { scoreItem, scoreAll, HOT_KEYWORDS };
    ```

    **Cách tính điểm:**
    - Hot keywords: ai, gpt, trump, putin, biden... (40 điểm)
    - Số lượng cùng topic (30 điểm)
    - Engagement giả lập (30 điểm)

    const DEFAULT_TOP = 5;
    const MIN_SCORE = 20;

    function rankNews(scoredItems, options = {}) {
      const top = options.top || DEFAULT_TOP;
      const minScore = options.minScore || MIN_SCORE;

      // Sort theo score giảm dần
      const sorted = scoredItems
        .filter(i => i.hotScore >= minScore)
        .sort((a, b) => b.hotScore - a.hotScore);

      // Lấy top N
      return sorted.slice(0, top);
    }

    function getHotNews(items, options = {}) {
      const top = options.top || DEFAULT_TOP;

      // Score all
      const scored = items.map(item => ({
        ...item,
        hotScore: calculateScore(item)
      }));

      // Rank
      return rankNews(scored, { top });
    }

    function calculateScore(item) {
      let score = 0;

      const titleLower = item.title.toLowerCase();
      const summaryLower = item.summary?.toLowerCase() || '';

      const hotKeywords = [
        'AI', 'GPT', 'OpenAI', 'Google', 'Apple', 'Meta', 'Microsoft',
        'iPhone', 'Samsung', 'Tesla', 'Trump', 'Putin', 'Biden',
        'Nvidia', 'chip', 'Semicon', 'tariffs', 'trade war',
        'Bitcoin', 'Crypto', 'USD', 'Fed', 'interest rate'
      ];

      for (const kw of hotKeywords) {
        if (titleLower.includes(kw.toLowerCase())) score += 10;
      }

      if (item.summary?.length > 100) score += 5;
      if (/\d+/.test(titleLower)) score += 5;
      if (item.source === 'BBC' || item.source === 'Reuters') score += 10;

      return score;
    }

    module.exports = { rankNews, getHotNews, DEFAULT_TOP, MIN_SCORE };
    ```

    **Config:**
    - Default lấy top 5 tin hot nhất
    - Minimum score để qualify: 20 điểm
