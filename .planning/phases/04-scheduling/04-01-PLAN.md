    interface LogEntry {
      timestamp: string;
      level: LogLevel;
      message: string;
      data?: Record<string, unknown>;
    }

    function log(level: LogLevel, message: string, data?: Record<string, unknown>): void {
      const entry: LogEntry = {
        timestamp: new Date().toISOString(),
        level,
        message,
        ...(data && { data })

      console.log(JSON.stringify(entry));
    }

    2. **src/services/scheduler.ts**:
    ```typescript
    import { log } from './logger';
    import { checkAndAlert } from './alert';

    // In-memory failure counter (resets on worker restart)
    let failureCount = 0;
    const MAX_RETRIES = 3;

    async function runPipeline(): Promise<{ success: boolean; message: string }> {
      const runId = Date.now();
      
      try {
        log('info', `Pipeline run ${runId} started`, { runId });
        
        // Import and run the feed pipeline (Phase 2 + Phase 3)
        // Note: Import should work once phases are implemented
        const { fetchAndPost } = await import('./feed').catch(() => null);
        
        if (fetchAndPost) {
          await fetchAndPost();
        } else {
          // Placeholder if phases not implemented yet
          log('warn', 'Feed pipeline not implemented yet', { runId });
        }
        
        // Success - reset failure counter
        failureCount = 0;
        log('info', `Pipeline run ${runId} completed successfully`, { runId });
        
        return { success: true, message: 'Pipeline completed' };
        
      } catch (error) {
        failureCount++;
        const errorMessage = error instanceof Error ? error.message : String(error);
        
        log('error', `Pipeline run ${runId} failed`, { 
          runId, 
          error: errorMessage,
          failureCount 
        });
        
        // Check if we should alert
        if (failureCount >= MAX_RETRIES) {
          await checkAndAlert(failureCount, errorMessage);
        }
        
        return { success: false, message: errorMessage };
      }
    }

    function getFailureCount(): number {
      return failureCount;
    }

    function resetFailureCount(): void {
      failureCount = 0;
    }

    export { runPipeline, getFailureCount, resetFailureCount };
    ```

    3. **src/middleware/cron.ts** (cron trigger handler):
    ```typescript
    import { Hono } from 'hono';
    import { runPipeline } from '../services/scheduler';
    import { log } from '../services/logger';

    const cronRoute = new Hono();

    // Cloudflare Cron Trigger handler
    // Called every hour at :00 (with random delay 0-15 min configured separately)
    cronRoute.post('/cron', async (c) => {
      const startTime = Date.now();
      
      // Add random delay to avoid spam detection (0-15 minutes)
      const delayMs = Math.floor(Math.random() * 15 * 60 * 1000);
      await new Promise(resolve => setTimeout(resolve, delayMs));
      
      const result = await runPipeline();
      const duration = Date.now() - startTime;
      
      return c.json({
        success: result.success,
        message: result.message,
        durationMs: duration,
        timestamp: new Date().toISOString()
      });
    });

    export { cronRoute };
    ```

    Note: The handler adds delay internally since Cloudflare cron doesn't support delay configuration.

  <action>
    Create Telegram alerting service:

    1. **src/services/alert.ts**:
    ```typescript
    import { log } from './logger';

    interface Env {
      TELEGRAM_BOT_TOKEN: string;
      TELEGRAM_CHAT_ID: string;
    }

    const API_URL = 'https://api.telegram.org/bot';

    async function sendTelegramAlert(message: string, env: Env): Promise<boolean> {
      if (!env.TELEGRAM_BOT_TOKEN || !env.TELEGRAM_CHAT_ID) {
        log('warn', 'Telegram not configured - skipping alert');
        return false;
      }

      const url = `${API_URL}${env.TELEGRAM_BOT_TOKEN}/sendMessage`;
      
      const payload = {
        chat_id: env.TELEGRAM_CHAT_ID,
        text: message,
        parse_mode: 'Markdown'
      };

      try {
        const response = await fetch(url, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(payload)
        });

        if (!response.ok) {
          const error = await response.text();
          log('error', 'Telegram alert failed', { error, status: response.status });
          return false;
        }

        log('info', 'Telegram alert sent successfully', { message: message.substring(0, 50) });
        return true;
        
      } catch (error) {
        const errorMessage = error instanceof Error ? error.message : String(error);
        log('error', 'Telegram alert error', { error: errorMessage });
        return false;
      }
    }

    async function checkAndAlert(failureCount: number, lastError: string): Promise<void> {
      // This function is called by scheduler when failureCount >= 3
      // Environment is passed via global or context
      
      const message = `🚨 *TongHopTinTuc Alert*\n\n` +
        `*Error:* ${lastError}\n` +
        `*Time:* ${new Date().toISOString()}\n` +
        `*Count:* ${failureCount} consecutive failures\n\n` +
        `*Action needed!*`;

      // Get env from context - this will be injected by the runtime
      // In Cloudflare Workers, access via c.env
      const env = globalThis as unknown as { env?: Env };
      
      if (env.env?.TELEGRAM_BOT_TOKEN && env.env?.TELEGRAM_CHAT_ID) {
        await sendTelegramAlert(message, env.env);
      }
    }

    export { sendTelegramAlert, checkAndAlert, type Env };
    ```
